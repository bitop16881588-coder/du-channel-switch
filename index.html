const SHEET = 'overrides'; function doGet(e) { return handle(e, 'get'); } function doPost(e) { return handle(e, 'post'); } function handle(e, method) { const lock = LockService.getScriptLock(); try { lock.waitLock(20000); } catch (err) { return out({ ok: false, error: 'busy' }); } try { const sh = getSheet(); let action = 'pull', body = {}; if (method === 'post' && e.postData && e.postData.contents) { body = JSON.parse(e.postData.contents); action = body.action || 'push'; } else if (e.parameter && e.parameter.action) { action = e.parameter.action; } if (action === 'pull') { const rows = sh.getDataRange().getValues(); rows.shift(); const cells = rows .filter(function (r) { return r[0] !== '' && r[0] !== null; }) .map(function (r) { return { i: String(r[0]), f: String(r[1]), v: String(r[2]), t: Number(r[3]) || 0 }; }); return out({ ok: true, cells: cells }); } if (action === 'push') { const cells = body.cells || []; const rows = sh.getDataRange().getValues(); const idx = {}; for (var r = 1; r < rows.length; r++) { idx[rows[r][0] + '|' + rows[r][1]] = r + 1; } cells.forEach(function (c) { const key = c.i + '|' + c.f; const ts = Number(c.t) || 0; const rn = idx[key]; if (rn) { const existTs = Number(sh.getRange(rn, 4).getValue()) || 0; if (ts >= existTs) { sh.getRange(rn, 1, 1, 4).setValues([[c.i, c.f, c.v, ts]]); } } else { sh.appendRow([c.i, c.f, c.v, ts]); idx[key] = sh.getLastRow(); } }); return out({ ok: true, saved: cells.length }); } return out({ ok: false, error: 'unknown action' }); } catch (err) { return out({ ok: false, error: String(err) }); } finally { lock.releaseLock(); } } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(SHEET); if (!sh) { sh = ss.insertSheet(SHEET); sh.appendRow(['id', 'field', 'value', 'ts']); } return sh; } function out(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
